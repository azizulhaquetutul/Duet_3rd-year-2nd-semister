#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main() {
    int head = 53;
    vector<int> disk = {12, 34, 52, 14, 25, 68, 39};
    int size = 90;
    int sum = 0;

    // Add head to list and sort
    disk.push_back(head);
    sort(disk.begin(), disk.end());

    // Find position of head in sorted list
    int pos = find(disk.begin(), disk.end(), head) - disk.begin();

    cout << "The position of head:\n";
    cout << head;

    // Move right (towards end)
    for (int i = pos + 1; i < disk.size(); i++) {
        cout << "->" << disk[i];
        sum += abs(disk[i] - head);
        head = disk[i];
    }

    // Go to end of disk if needed
    if (head != size) {
        cout << "->" << size;
        sum += abs(size - head);
        head = size;
    }

    // Jump from end to beginning
    cout << "->0";
    sum += abs(head - 0);
    head = 0;

    // Move from beginning to just before head position
    for (int i = 0; i < pos; i++) {
        cout << "->" << disk[i];
        sum += abs(disk[i] - head);
        head = disk[i];
    }

    cout << "\nTotal head Movements (C-SCAN): " << sum << endl;
    return 0;
}
