#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;  cin >> n;
    int res; cin >> res;
    vector<int>Allocation[n],Need[n],Available(n);
    int x;
    for(int i=0; i<n; i++){
        for(int j=0; j<res; j++){
            cin >> x;
            Allocation[i].push_back(x);
        }
    }
    for(int i=0; i<n; i++){
        for(int j=0; j<res; j++){
            cin >> x;
            Need[i].push_back(x);
        }
    }
    for(int i=0; i<res; i++){
        cin >> Available[i];
    }
    vector<int>Current_need[n];
    for(int i=0; i<n; i++){
        for(int j=0; j<res; j++){
            x = Need[i][j]-Allocation[i][j];
            if(x<0) x = 0;
            Current_need[i].push_back(x);
        }
    }
    bool ok[n] = {false};
    int cunt = 0, k = 0;
    vector<int>ans;
    while(cunt<n){
        bool found = false;
        for(int i=0;i<n;i++){
            if(!ok[i]){
                bool canAllocate=true;
                for(int j=0;j<res;j++){
                    if(Available[j]<Current_need[i][j]){
                        canAllocate=false;
                        break;
                    }
                }
                if(canAllocate){
                    cunt++;
                    for(int j=0;j<res;j++){
                        Available[j]+=Allocation[i][j];
                    }
                    ans.push_back(i);
                    ok[i] = true;
                    found = true;

                }

            }
        }
        if(!found){
            cout << "System is in an unsafe state.\n";
            return 0;
        }
    }
    cout << "System is in a safe state.\nSafe sequence is: ";
    for(auto it:ans) cout << 'P' << it  << ' ';
}
